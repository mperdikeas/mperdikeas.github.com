  2. loci for the yield and yield! operators
    3. Acceptable context for the yield and yield! operators are sequences (seq {}), lists ([]) and arrays ([| |])
  3. F# seq, yield and yield! operators
    4. open System

let main=
    seq {
        for i in 1..10 do
            yield i
        let i = [11;12;13;14;15]
        yield! i
    }


main |> Seq.toList |> List.iter ( fun n -> printfn "%d" n ) ;;
